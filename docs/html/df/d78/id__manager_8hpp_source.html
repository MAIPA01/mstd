38 _MSTD_CONSTEXPR20 void _update_removed_ids() {
39 if (_removedIds.empty()) { return; }

41 auto last = std::prev(_removedIds.end());

43 --_nextId;
44 _removedIds.erase(*last);

61 [[nodiscard]] _MSTD_CONSTEXPR20 id_type get_next_id() {
62 if (!_removedIds.empty()) {
63 const id_type id = *_removedIds.begin();
64 _removedIds.erase(id);
65 return id;
66 }

75 _MSTD_CONSTEXPR20 bool return_id(id_type id) {
76 #if _MSTD_HAS_CXX20

78 #else
79 if (id == bad_id() || id >= _nextId || _removedIds.find(id) != _removedIds.end()) {
80 #endif
81 return false;
82 }
83
84 _removedIds.insert(id);
85 _update_removed_ids();

87 }
