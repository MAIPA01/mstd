82 _MSTD_CONSTEXPR20 T& insert(const const_iterator& where, const value_type& value) {
83 if (!contains(value.first)) {
84 // insert at where and set value (after update iterators in map)
85 size_type whereOffset = std::distance(_orderedElements.cbegin(), where);
86
87 _orderedElements.insert(where, value);
88
89 _update_indexes(whereOffset);
90
91 return _orderedElements.at(_elementsMap.at(value.first)).second;
92 }
93
94 // move key to where and change value (after update iterators in map)
95 size_type whereOffset =
96 std::clamp<size_type>(std::distance(_orderedElements.cbegin(), where), 0, _orderedElements.size() - 1);
97 const size_type& elemOffset = _elementsMap.at(value.first);
98
99 // remove elem
100 _orderedElements.erase(std::next(_orderedElements.begin(), elemOffset));
101
102 // insert new elem
103 _orderedElements.insert(std::next(_orderedElements.begin(), whereOffset), value);
104
105 // update iterators in map
106 if (whereOffset > elemOffset) { _update_indexes(elemOffset); }
107 else { _update_indexes(whereOffset); }
108
109 return _orderedElements.at(_elementsMap.at(value.first)).second;
110 }

118 size_t currWhereOffset = std::distance(_orderedElements.cbegin(), where);
119 for (Iter iter = begin; iter != end; ++iter, ++currWhereOffset) {
