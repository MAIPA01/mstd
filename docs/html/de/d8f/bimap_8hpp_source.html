83 _MSTD_CONSTEXPR20 T& insert(const value_type& value) {
84 if (contains(value.first) && _data[_map.at(value.first)].second == value.second) {
85 return _data[_map.at(value.first)].second;
86 }
87
88 // add new pair
89 _data.push_back(value);
90
91 // get maps iterators
92 size_t minIdx = _data.size() - 1;
93 if (contains(value.first)) {
94 size_t valueIdx = _map.at(value.first);
95
96 // set min idx
97 minIdx = std::min(minIdx, valueIdx);
98
99 // erase inverted map
100 _invertedMap.erase(_data[valueIdx].second);
101
102 // erase data
103 _data.erase(std::next(_data.begin(), valueIdx));
104
105 // erase map
106 _map.erase(value.first);
107 }
108
109 if (contains_value(value.second)) {
110 size_t keyIdx = _invertedMap.at(value.second);
111
112 // set min idx
113 minIdx = std::min(keyIdx, minIdx);
114
115 // erase map
116 _map.erase(_data[keyIdx].first);
117
118 // erase data
119 _data.erase(std::next(_data.begin(), keyIdx));
120
121 // erase inverted map
122 _invertedMap.erase(value.second);
123 }
124
125 _update_indexes(minIdx);
126
127 return _data[_map.at(value.first)].second;
128 }

140 if (!contains(key)) { return; }
