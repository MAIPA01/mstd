50 _MSTD_CONSTEXPR20 void _append_indexes(size_type count) {
51 size_type newSize = _dataIndex.size() + count;
52
53 _id.reserve(newSize);
54 _dataIndex.reserve(newSize);
55
56 for (size_type i = _dataIndex.size(); i < newSize; ++i) {
57 _dataIndex.push_back(i);
58 _id.push_back(i);
59 }
60 }

122 _MSTD_CONSTEXPR20 iterator insert_at(size_type id, const T& value) {
123 if (has_value(id)) {
124 _data[_dataIndex[id]] = value;
125 return std::next(_data.begin() + _dataIndex[id]);
126 }
127
128 _append_indexes(id + 1 - size());
129
130 // insert data
131 auto iter = _data.insert(_data.cend(), value);
132
133 if (_data.size() != size()) {
134 size_type toSwapId = _id[_data.size() - 1];
135
136 // swap ids
137 std::swap(_id[_data.size() - 1], _id[id]);
138
139 // swap indexes
140 std::swap(_dataIndex[toSwapId], _dataIndex[id]);
141 }
142
143 return iter;
144 }

259 _MSTD_CONSTEXPR20 iterator erase(size_type id) {
260 mstd_assert(id < size(), "Index out of bounds");
261 if (!has_value(id)) { return _data.end(); }
262
263 // get data index
264 size_t index = _dataIndex[id];
265 size_t lastIndex = _data.size() - 1;
266
267 // swap data
268 std::swap(_data[index], _data[lastIndex]);
269
270 // swap ids
271 std::swap(_id[index], _id[lastIndex]);
272
273 // erase last item
274 auto itr = _data.erase(std::prev(_data.cend()));
275
276 // update indexes
277 std::swap(_dataIndex[_id[index]], _dataIndex[id]);
278
279 // change size if it is possible
280 if (_dataIndex[lastIndex] == lastIndex) {
281 _dataIndex.erase(std::prev(_dataIndex.cend()));
282 _id.erase(std::prev(_id.cend()));
283 }
284
285 return itr;
286 }
