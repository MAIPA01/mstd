78 _MSTD_CONSTEXPR20 T& insert(const const_iterator& where, const T& item) {
79 if (!contains(item)) {
80 size_t whereOffset =
81 std::clamp<size_t>(std::distance(_orderedElements.cbegin(), where), 0, _orderedElements.size());
82
83 _orderedElements.insert(where, item);
84
85 _update_indexes(whereOffset);
86 return *std::next(_orderedElements.begin(), whereOffset);
87 }
88
89 size_t whereOffset =
90 std::clamp<size_t>(std::distance(_orderedElements.cbegin(), where), 0, _orderedElements.size() - 1);
91 const size_t& elementOffset = _elementsMap.at(item);
92
93 // remove element
94 _orderedElements.erase(std::next(_orderedElements.cbegin(), elementOffset));
95
96 // insert element where
97 _orderedElements.insert(std::next(_orderedElements.cbegin(), whereOffset), item);
98
99 // update iterators
100 if (whereOffset > elementOffset) { _update_indexes(elementOffset); }
101 else { _update_indexes(whereOffset); }
102
103 return *std::next(_orderedElements.begin(), whereOffset);
104 }
