58 template<class F, class Ret, class... Args>
59 struct base_function_view<F, Ret, std::tuple<Args...> > {
60 private:

63 utils::is_valid_free_function_v<FreeFunc, F, Ret, Args...>;

72 template<class FreeFunc>
73 static Ret _free_func_invoker(void* func, void*, Args&&... args) noexcept(is_noexcept_function_v<FreeFunc>) {
74 using func_ptr = c_func_t<function_type_t<FreeFunc> >;
75
76 func_ptr freeFunc = reinterpret_cast<func_ptr>(func);
77 return std::invoke(freeFunc, std::forward<Args>(args)...);
78 }
79
80 template<class MemberFunc>
81 static Ret _member_func_invoker(void* func, void* parent,
82 Args&&... args) noexcept(is_noexcept_function_v<MemberFunc>) {
83 using parent_type = function_parent_t<MemberFunc>;
84 using parent_ptr = parent_type*;
85 using func_ptr = c_member_func_t<parent_type, function_type_t<MemberFunc> >;
86
87 parent_ptr parentObj = static_cast<parent_type*>(parent);
88 func_ptr memberFunc = *reinterpret_cast<const func_ptr*>(func);
89 return std::invoke(memberFunc, parentObj, std::forward<Args>(args)...);
90 }
91
92 template<class Functor>
93 static Ret _functor_invoker(void* func, void*, Args&&... args) noexcept(is_noexcept_function_v<Functor>) {
94 return std::invoke(*static_cast<Functor*>(func), std::forward<Args>(args)...);
95 }
96
97 using _invoker_type = Ret (*)(void*, void*, Args&&...) noexcept(is_noexcept_function_v<F>);
98
99 void* _func = nullptr;
100 void* _parent = nullptr;
101 _invoker_type _invoker = nullptr;
102
103 public:
104 base_function_view() noexcept = default;
105 base_function_view(const base_function_view&) noexcept = default;
106
107 base_function_view(base_function_view&& other) noexcept
108 : _func(std::exchange(other._func, nullptr)), _parent(std::exchange(other._parent, nullptr)),
109 _invoker(std::exchange(other._invoker, nullptr)) {}
110
111 virtual ~base_function_view() noexcept = default;
112
113 base_function_view& operator=(const base_function_view&) noexcept = default;
114
115 base_function_view& operator=(base_function_view&& other) noexcept {
116 _func = std::exchange(other._func, nullptr);
117 _parent = std::exchange(other._parent, nullptr);
118 _invoker = std::exchange(other._invoker, nullptr);
119 return *this;
120 }
