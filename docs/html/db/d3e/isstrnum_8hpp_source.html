61 inline bool isstrnum(const std::string_view str) {
62 if (str.empty()) { return false; }
63
64 if (str.size() > 2 && str[0] == '0') {
65 if (str[1] == 'b') { return isstrbin(str); }
66
67 if (str[1] == 'c') { return isstroct(str); }
